    ]),

# Function to generate a new UUID
def generate_id():
    return str(uuid.uuid4())

# Generate SQL statements
sql_statements = []
for parent_id, (parent_name, subcategories) in categories.items():
    parent_sql = f"INSERT INTO \"categories\" (`category_id`, `category_name`, `parent_category_id`) VALUES ('{parent_id}', '{parent_name}', '0');"
    sql_statements.append(parent_sql)
    for subcategory in subcategories:
        sub_id = generate_id()
        sub_sql = f"INSERT INTO \"categories\" (`category_id`, `category_name`, `parent_category_id`) VALUES ('{sub_id}', '{subcategory}', '{parent_id}');"
        sql_statements.append(sub_sql)

# Save the SQL statements to a text file
with open('c:\\temp\\insert_categories.sql', 'w') as file:
    for statement in sql_statements:
        file.write(statement + '\n')

print("SQL insert statements have been saved to insert_categories.sql")
